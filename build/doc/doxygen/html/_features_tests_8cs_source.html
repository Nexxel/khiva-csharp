120  }
121  }

123 
124  private void Flatten3D<T>(ref T[] flattenArr, T[,,] arr)
125  {
126  for (int i = 0; i < arr.GetLength(0); i++)
127  {
128  for (int j = 0; j < arr.GetLength(1); j++)
129  {
130  for (int k = 0; k < arr.GetLength(2); k++)
131  {
132  flattenArr[i * (arr.GetLength(1) * arr.GetLength(2)) + j * arr.GetLength(2) + k] = arr[i, j, k];
133  }
134  }
135  }
